110 int Config::getCFG()
111 {
112  int rt = 0;
113  //先清空链表
114  freeJoin();
115  //配置总数为0
116  this->cfg_line = 0;
117  //定义缓存字符串变量
118  string st;
119  string key, value;
120  //定义文件变量
121  ifstream infile;
122  string::size_type idx;
123  char *p = NULL, *q = NULL;
124  //打开文件
125  infile.open(this->file_name);
126  //遍历直到文件的最后
127  while (getline(infile,st))
128  {
129  //初始化缓存
130  //取得一行配置
131  //找不到等号则继续
132  idx = st.find("#");
133  if (idx != string::npos)
134  {
135  st = st.substr(0, idx);
136  }
137  idx = st.find("=");
138  if (idx == string::npos)
139  {
140  continue;
141  }
142 
143  //截断字符串得到key和value字符串
144  key = st.substr(0, idx);
145  key.erase(0,key.find_first_not_of(" "));
146  key.erase(key.find_last_not_of(" ")+1);
147  value = st.substr(idx + 1,st.length()-idx);
148  value.erase(0,value.find_first_not_of(" "));
149  value.erase(value.find_last_not_of(" ")+1);
150  // cout<<st<<"_________________-;"<<key<<":"<<value<<endl;
151  //插入链表
152  joinHead(key,value);
153  st = "";
154 
155  }
156  //关闭文件
157  infile.close();
158  return rt;

162 int Config::joinHead(string key,string value)
163 {
164  int rt = 1;
165  if (this->head == NULL)
166  {
167  rt = 2;
168  return rt;
169  }
170  //定义移动指针
171  CFG_J * p = this->head;
172  CFG_J * cur = p->next;
173  while (cur != NULL)
174  {
175  //cout << cur->key << " " << key << " " << cur->key.compare(key) << endl;
176  //找到值则直接改变
177  if (cur->key.compare(key) == 0)
178  {
179  cur->value = value;
180  rt = 0;
181  break;
182  }
183  p = cur;
184  cur = p->next;
185  }
186  //找不到值则再最后插入
187  if (rt != 0)
188  {
189  CFG_J *q = new CFG_J();
190  q->key = key;
191  q->value = value;
192  q->next = NULL;
193  p->next = q;
194  //配置数自增
195  this->cfg_line ++;
196  }
197  return rt;
198 }
